    • Javascript I.: by val/by ref

    JavaScript Objects (and by extension: arrays, regexes, dates, etc.) equality and assignment are reference based: 
    {a:'a'} == {a:'a'} // false
    But: 
    var myObject = {a:'a'};
var myObject2 = myObject;

myObject == myObject2 // true

    Furthermore: 

    myObject.b = 'b';

console.log(myObject2.b); // Logs: "b"

    Javascript 1.: scope

    A var statement

    var a = 1;
function hello() {
  a = 2
}
hello();
console.log(a); // -> 2

    Block scope

    function example() {
  var o = {}, i = 0;
  for (i = 0; i < 3; i++) {
    o[i] = function() { console.log(i); };
  }
  o[0]();
  o[1]();
  o[2]();
}

example(); // -> 3, 3, 3

    function example() {
  var o = {};

  [0, 1, 2, 3].forEach(function(i) {
    o[i] = function() { console.log(i); };
  });

  o[0]();
  o[1]();
  o[2]();
}

example(); // -> 1, 2, 3

    • Javascript I.: objects 2

    this

    Művelet vagy függvény? Egy objektum tulajdonságaként megadott függvény a művelet.

    var myobj = {
  log_this: function() {console.log(this);}
}
var MyConst = function() {
  this.log_this: function() {console.log(this);} 
}
myobj.log_this(); // => window
myObj = new MyConst(); 
myObj.log_this(); // => myObj
window.setTimeout(myObj.log_this, 100); // => window
    if a function is called as a method on an object, or by using new, then the value of this will be the current instance of the object, otherwise it will be the global context
  • Dailjs cikk a this-ről és a bindingról

    • Javascript I.: objects 3

    function Animal() {
}
console.log(Animal.prototype)

Animal.prototype.type = 'Unknown';
Animal.prototype.weight = 0;
Animal.prototype.weightUnits = 'kg';

Animal.prototype.toString = function() {
  return this.type + ', ' + this.weight + this.weightUnits;
};

var molly = new Animal();
molly.type = 'Dog';
molly.weight = 28;

console.log(molly.toString()); // -> "Dog, 28kg"
  • Az öröklődés a prototype-n keresztül történik.
